# TP3JeuxVideo

IMPORTANT !!!!!!!!!!!!!!!! Seulement la scène de jeu peut utiliser les manettes


Pour jouer avec clavier souris :
	joueur1(bleu) =
		gauche = a
		droite = d
		sauter = espace
		tirer = click gauche
		tireBonus = click droite
	joueur2(rouge) =
		gauche = <-
		droite = ->
		sauter = flèche du haut
		tirer = '
	viseur = souris

Pour jouer avec une manette :
	joueur1(bleu) et joueur 2(Rouge) =
		déplacements = gachette gauche
		viseur = gachette droite
		sauter = a
		tirer = RT
		tireBonus = RB
